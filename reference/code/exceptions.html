    def __init__(self, cmd: list[str], stdout: str = "", stderr: str = ""):
    """
    Args:
        cmd: failed command.
        stdout: standard output of that process.
        stderr: standard error of that process.
    """
    self.cmd = cmd
    self.stdout = stdout
    self.stderr = stderr

    super().__init__(
        "The following process exited with a non-zero error:\n"
        f"===  CMD   ===\n{' '.join(cmd)}\n============\n"
        f"=== STDOUT ===\n{stdout}============\n"
        f"=== STDERR ===\n{stderr}============\n"
    )
